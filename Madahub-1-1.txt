-- Click Animation
local function brighten(color, delta)
    return Color3.fromRGB(
        math.clamp(math.floor(color.R * 255 + delta), 0, 255),
        math.clamp(math.floor(color.G * 255 + delta), 0, 255),
        math.clamp(math.floor(color.B * 255 + delta), 0, 255)
    )
end

local function applyClickAnimation(btn)
    local originalSize = btn.Size
    local originalColor = btn.BackgroundColor3
    local originalTransparency = btn.BackgroundTransparency

    btn.AutoButtonColor = false

    local hoverStroke = btn:FindFirstChildOfClass("UIStroke")
    if not hoverStroke then
        hoverStroke = Instance.new("UIStroke")
        hoverStroke.Thickness = 1
        hoverStroke.Transparency = 0.85
        hoverStroke.Color = Color3.fromRGB(255, 255, 255)
        hoverStroke.Parent = btn
    end

    btn.MouseEnter:Connect(function()
        TweenService:Create(btn, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            BackgroundColor3 = brighten(originalColor, 12),
            BackgroundTransparency = math.max(0, originalTransparency - 0.06)
        }):Play()
        TweenService:Create(hoverStroke, TweenInfo.new(0.12), {Transparency = 0.65}):Play()
    end)

    btn.MouseLeave:Connect(function()
        TweenService:Create(btn, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            BackgroundColor3 = originalColor,
            BackgroundTransparency = originalTransparency
        }):Play()
        TweenService:Create(hoverStroke, TweenInfo.new(0.12), {Transparency = 0.85}):Play()
    end)

    btn.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            TweenService:Create(btn, TweenInfo.new(0.08), {
                Size = UDim2.new(originalSize.X.Scale, originalSize.X.Offset - 2, originalSize.Y.Scale, originalSize.Y.Offset - 2),
                BackgroundColor3 = brighten(originalColor, 20)
            }):Play()
        end
    end)
    btn.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            TweenService:Create(btn, TweenInfo.new(0.08), {
                Size = originalSize,
                BackgroundColor3 = brighten(originalColor, 12)
            }):Play()
        end
    end)
end

local SymbolData = {
    ["Smart"] = "SMART_FLAG",
    ["EUR^"] = string.rep("EUR^", 50),
    ["PI_"] = string.rep("PI_", 50),
    ["TM_COPY_REG"] = string.rep("_TM_COPY_REG_", 15),
    ["SECTION~"] = string.rep("SECTION~", 25),
    ["OMEGA_"] = string.rep("OMEGA_", 35),
    ["@@"] = string.rep("@@", 170),
    ["__"] = string.rep("__", 160)
}

local AvailableSymbols = {}
local lastSmartSymbol = ""

local function refillSymbols()
    AvailableSymbols = {}
    for key, val in pairs(SymbolData) do
        if key ~= "Smart" then
            table.insert(AvailableSymbols, val)
        end
    end
end
refillSymbols()

local function getSmartSymbol()
    if #AvailableSymbols == 0 then refillSymbols() end
    local randomIndex = math.random(1, #AvailableSymbols)
    local chosen = AvailableSymbols[randomIndex]
    if chosen == lastSmartSymbol and #AvailableSymbols > 1 then
        randomIndex = (randomIndex % #AvailableSymbols) + 1
        chosen = AvailableSymbols[randomIndex]
    end
    lastSmartSymbol = chosen
    return chosen
end

local function ForcePlayMusic(Remote, ID)

    Remote:FireServer("PickingScooterMusicText", tostring(numericId))
    return true
end

-- Chat Logic
local function sendToRobloxChat(message)
    local now = os.clock()
    if now - lastChatSentAt < 0.35 then
        return
    end
    lastChatSentAt = now

    if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        local channel = TextChatService.TextChannels:FindFirstChild("RBXGeneral")
        if channel then channel:SendAsync(message) end
    else
        local sayEvent = ReplicatedStorage:FindFirstChild("DefaultChatSystemChatEvents")
        if sayEvent and sayEvent:FindFirstChild("SayMessageRequest") then
            sayEvent.SayMessageRequest:FireServer(message, "All")
        end
    end
end

local function sendSafeSpam(prefixName, customText)
    local prefixString = prefixName == "Smart" and getSmartSymbol() or (SymbolData[prefixName] or "")
    local finalMessage = customText ~= "" and prefixString .. " " .. customText or prefixString
    sendToRobloxChat(finalMessage)
end

                    BackgroundColor3 = Color3.fromRGB(24, 28, 38),
                    BackgroundTransparency = 0.35,
                    TextColor3 = Color3.fromRGB(160, 170, 190)
                }):Play()

                if childStroke then
                    TweenService:Create(childStroke, TweenInfo.new(0.2), {Transparency = 0.9}):Play()

local function playMusicEverywhere(id)
    local numericId = tostring(id or "")
    if numericId == "" then
        return
    end
    local re = ReplicatedStorage:FindFirstChild("RE")
    if not re then
        return
    end

    pcall(function()
        local ev = re:FindFirstChild("PlayerToolEvent")
        if ev then ev:FireServer("ToolMusicText", numericId) end
    end)
    pcall(function()
        local ev = re:FindFirstChild("1Player1sHous1e")
        if ev then ev:FireServer("PickHouseMusicText", numericId) end
    end)
    pcall(function()
        local ev = re:FindFirstChild("1Player1sCa1r")
        if ev then ev:FireServer("PickingCarMusicText", numericId) end
    end)
    pcall(function()
        local ev = re:FindFirstChild("1NoMoto1rVehicle1s")
        if ev then ev:FireServer("PickingScooterMusicText", numericId) end
    end)
end

local function safeLoadRemote(url)
    local ok, err = pcall(function()
        local src = game:HttpGet(url, true)
        local fn = loadstring(src)
        if fn then
            fn()
        end
    end)
    if not ok then
        warn("Failed to load remote script:", err)
    end
end

local function forceMinZIndex(root, minZ)
    if root:IsA("GuiObject") then
        root.ZIndex = math.max(root.ZIndex, minZ)
    end
    for _, obj in ipairs(root:GetDescendants()) do
        if obj:IsA("GuiObject") then
            obj.ZIndex = math.max(obj.ZIndex, minZ)
        end
    end
end

local function startPropEngine(mode, targetName, visType)
    stopPropEngine()
    task.wait(0.1) 
    
    activeMode = mode
    engineLoopActive = true
    
    task.spawn(function()
        while scriptAlive and engineLoopActive do
            local root = getTargetRoot(targetName)
            local props = getMyProps()
            
            if root and #props > 0 then
                local t = tick()
                
                for i, prop in ipairs(props) do
                    local baseCFrame = root.CFrame
                    local finalCFrame = baseCFrame
                    local color = Color3.fromHSV((t * 0.5 + (i/#props)) % 1, 1, 1) 
                    
                    if activeMode == "Nuke" then
                        finalCFrame = baseCFrame * CFrame.new(math.random(-2,2), math.random(0,3), math.random(-2,2)) 
                            * CFrame.Angles(math.random(), math.random(), math.random())
                    
                    elseif activeMode == "Visualizer" then
                        if visType == "Orbital" then
                            local angle = (t * 3) + ((i / #props) * math.pi * 2)
                            finalCFrame = baseCFrame * CFrame.new(math.cos(angle) * 6, 0, math.sin(angle) * 6)
                        elseif visType == "Tornado" then
                            local angle = (t * 5) + ((i / #props) * math.pi * 2)
                            local heightOffset = (i / #props) * 12
                            local radius = 2 + (i / #props) * 5
                            finalCFrame = baseCFrame * CFrame.new(math.cos(angle) * radius, heightOffset - 4, math.sin(angle) * radius)
                        elseif visType == "Swarm" then
                            local offsetX = math.noise(t, i, 0) * 8
                            local offsetY = math.noise(t, 0, i) * 6
                            local offsetZ = math.noise(0, t, i) * 8
                            finalCFrame = baseCFrame * CFrame.new(offsetX, offsetY + 2, offsetZ)
                        elseif visType == "Grid" then
                            local spacing = 5
                            local cols = math.ceil(math.sqrt(#props))
                            local row = math.floor((i-1) / cols)
                            local col = (i-1) % cols
                            finalCFrame = baseCFrame * CFrame.new((col - cols/2)*spacing, 0, (row - cols/2)*spacing)
                        end
                    end
                    
                    pcall(function()
                        prop.SetCurrentCFrame:InvokeServer(finalCFrame)
                        if prop:FindFirstChild("ChangePropColor") then
                            prop.ChangePropColor:InvokeServer(color)
                        end
                    end)
                end
            end
            task.wait(0.12) 
        end
    end)
end

local ProtectionStates = {
    AntiSit = false,
    Noclip = false,
    SuperNoclip = false,
    AntiFling = false,
    Freeze = false
}

local Cons = {}

local function toggleProtection(feature, state)
    ProtectionStates[feature] = state
    local char = LocalPlayer.Character
    local hum = char and char:FindFirstChildOfClass("Humanoid")
    local root = char and char:FindFirstChild("HumanoidRootPart")

    if Cons[feature] then
        Cons[feature]:Disconnect()
        Cons[feature] = nil
    end

    if feature == "AntiSit" then
        if state then
            if hum and hum.Sit then hum.Sit = false end
            Cons.AntiSit = RunService.Heartbeat:Connect(function()
                local c = LocalPlayer.Character
                local h = c and c:FindFirstChildOfClass("Humanoid")
                if h and h.Sit then h.Sit = false end
            end)
        end

    elseif feature == "Noclip" then
        if state then
            Cons.Noclip = RunService.Heartbeat:Connect(function()
                if LocalPlayer.Character then
                    for _, v in pairs(LocalPlayer.Character:GetDescendants()) do
                        if v:IsA("BasePart") and v.CanCollide then
                            if not string.find(v.Name:lower(), "leg") and not string.find(v.Name:lower(), "foot") then
                                v.CanCollide = false
                            end
                        end
                    end
                end
            end)
        end

    elseif feature == "SuperNoclip" then
        if state then
            Cons.SuperNoclip = RunService.Heartbeat:Connect(function()
                if LocalPlayer.Character then
                    for _, v in pairs(LocalPlayer.Character:GetDescendants()) do
                        if v:IsA("BasePart") and v.CanCollide then
                            v.CanCollide = false
                        end
                    end
                end
            end)
        end

    elseif feature == "AntiFling" then
        if state then
            local lastStabilize = 0
            Cons.AntiFling = RunService.Heartbeat:Connect(function()
                local myRoot = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                if myRoot and (os.clock() - lastStabilize) >= 0.1 then
                    lastStabilize = os.clock()
                    if myRoot.Velocity.Magnitude > 250 or myRoot.RotVelocity.Magnitude > 250 then
                        myRoot.Velocity = Vector3.new(0, 0, 0)
                        myRoot.RotVelocity = Vector3.new(0, 0, 0)
                    end
                end
            end)
        end

    elseif feature == "Freeze" then
        if root then root.Anchored = state end
        if state then
            Cons.Freeze = LocalPlayer.CharacterAdded:Connect(function(newChar)
                local newRoot = newChar:WaitForChild("HumanoidRootPart", 5)
                if newRoot then newRoot.Anchored = true end
            end)
        end
    end
end

clearProtectionState = function()
    for feature, con in pairs(Cons) do
        if con then
            con:Disconnect()
        end
        Cons[feature] = nil
    end
    for key, _ in pairs(ProtectionStates) do
        ProtectionStates[key] = false
    end
end

local function applyModernPolish(root)
    for _, obj in ipairs(root:GetDescendants()) do
        if obj:IsA("TextBox") then
            obj.BackgroundColor3 = Color3.fromRGB(26, 30, 40)
            obj.PlaceholderColor3 = Color3.fromRGB(140, 150, 170)
            local stroke = obj:FindFirstChildOfClass("UIStroke")
            if not stroke then
                stroke = Instance.new("UIStroke")
                stroke.Parent = obj
            end
            stroke.Color = Color3.fromRGB(255, 255, 255)
            stroke.Transparency = 0.88
            stroke.Thickness = 1

        elseif obj:IsA("ScrollingFrame") then
            obj.ScrollBarImageColor3 = Color3.fromRGB(125, 170, 255)
            obj.ScrollBarImageTransparency = 0.25

        elseif obj:IsA("TextLabel") then
            if obj.TextSize <= 10 then
                obj.TextSize = obj.TextSize + 1
            end

        elseif obj:IsA("TextButton") then
            if obj.TextSize <= 10 then
                obj.TextSize = obj.TextSize + 1
            end
        end
    end
end
